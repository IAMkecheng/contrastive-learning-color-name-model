python color-name-model.py train ncf_weight epoch_num name_encoder_layer_num device_id

ncf_weight: 0, 0.0001, 0.001, 0.01, 1
epoch_num: 30, 60
name_encoder_layer_num: 0, 1, 2 # 0表示投影层后接一个relu+线性层，1表示再加一个relu+线性层
device_id: 0, 1, 2, ...

# 测试name encoder加深是否有好处
python color-name-model.py train 0 30 0 0
python color-name-model.py train 0 30 1 1
python color-name-model.py train 0 30 2 2
